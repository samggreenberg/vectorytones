# Per-media-type dependencies.
# Each media type declares its own requirements in its subdirectory.
# To add a new media type, create vtsearch/media/<yourtype>/requirements.txt
# and add a -r line below pointing to it.
-r vtsearch/media/audio/requirements.txt
-r vtsearch/media/video/requirements.txt
-r vtsearch/media/image/requirements.txt
-r vtsearch/media/text/requirements.txt

# Per-importer dependencies.
# Each importer declares its own requirements in its subdirectory.
# To add a new importer, create vtsearch/datasets/importers/<name>/requirements.txt,
# then add a -r line to requirements-importers.txt.
-r requirements-importers.txt

# Per-exporter dependencies.
# Each exporter declares its own requirements in its subdirectory.
# To add a new exporter, create vtsearch/exporters/<name>/requirements.txt,
# then add a -r line to requirements-exporters.txt.
-r requirements-exporters.txt
